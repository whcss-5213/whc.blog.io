const getArray = <T>(value: T, times: number = 5): T[] => {

    return new Array(times).fill([value1, value2]);
};

Pick<Type, Keys> ​

通过从 Type 中选取一组属性 Keys（字符串字面或字符串字面的并集）来构造一个类型。

typescript
interface Todo {
    title: string;
    description: string;
    completed: boolean;
}

type TodoPreview = Pick<Todo, 'title' | 'completed'>;

type TodoPreview = {
    title: string;
    completed: boolean;
}

type MyPick<T, K extends keyof T> = {
    [key in K]: T[key];
};
